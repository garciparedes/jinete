    def __repr__(self):
        return self._print(dict(self))

    def _print(self, values):
        values = ", ".join(f"{key}={value}" for key, value in values.items())
        return f"{self.__class__.__name__}({values})"

    def _print_key_value(self, key: str, value: Any):
        return f"{key}={value}"
